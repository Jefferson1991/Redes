Guia de estudio — Router-on-a-Stick + CML Console Server

Carpeta Router-on-a-Stick · VLSM 192.168.50.0/24 · Conexion via CML Console Server (paramiko SSH).

Imprimir o PDF: Ctrl+P. Escala 85-90% si quieres menos hojas.

Ideas clave (examen)

  • Router-on-a-Stick: un solo cable trunk 802.1Q entre switch y router; el router usa subinterfaces (GigabitEthernet1.10) con encapsulation dot1Q <vlan>.
  • Topologia CML: R1 G1 ↔ SW1 Gi0/0 (trunk). SW1 Gi0/1 VLAN 10 (PC1), Gi0/2 VLAN 20 (PC2).
  • VLSM: una red base partida en subredes de distinto tamaño; cada VLAN tiene su gateway en el router.
  • Conexion: No hay Telnet/SSH directo al router. Se usa paramiko para SSH al CML Console Server y open /Lab/Nodo/0 para acceder a la consola serial.
  • Hosts Linux: Alpine Linux, IP con sudo su + ip addr add + ip route add default via.

main.py

SSH al CML Console Server, abre consola serial del equipo, envia comandos.

main.py
# Router-on-a-Stick — main.py
#
# Se conecta al CML Console Server por SSH (paramiko)
# y envia comandos a cada dispositivo por su consola serial.
#
# Flujo: Tu PC --SSH--> CML (10.10.20.161) --open--> consola del equipo
import paramiko
import time
from config import (
    CML_HOST, CML_USER, CML_PASS, ENABLE_SECRET,
    CONSOLA_R1, CONSOLA_SW1, CONSOLA_PC1, CONSOLA_PC2,
    COMANDOS_R1, COMANDOS_SW1, COMANDOS_PC1, COMANDOS_PC2,
)


def leer(shell, espera=1):
    time.sleep(espera)
    salida = ''
    while shell.recv_ready():
        salida += shell.recv(65535).decode('utf-8', errors='ignore')
    return salida


def enviar(shell, cmd, espera=0.5):
    shell.send(cmd + '\n')
    return leer(shell, espera)


def abrir_consola(ruta):
    """SSH al CML Console Server y abre la consola de un nodo."""
    ssh = paramiko.SSHClient()
    ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
    ssh.connect(CML_HOST, username=CML_USER, password=CML_PASS,
                look_for_keys=False, allow_agent=False)
    shell = ssh.invoke_shell(width=200, height=50)
    leer(shell, 2)
    enviar(shell, f'open {ruta}', 4)
    enviar(shell, '', 2)
    enviar(shell, '', 2)
    return ssh, shell


def configurar_cisco(ruta, comandos, nombre):
    """Configura un equipo Cisco (router o switch)."""
    print(f'\n{"="*50}')
    print(f'  Configurando: {nombre}')
    print(f'{"="*50}')
    ssh, shell = abrir_consola(ruta)

    # Modo enable (manda password solo si lo pide)
    salida = enviar(shell, 'enable', 2)
    if 'assword' in salida:
        enviar(shell, ENABLE_SECRET, 1)

    # Enviar comandos
    enviar(shell, 'configure terminal', 1)
    for cmd in comandos:
        enviar(shell, cmd, 0.3)
        print(f'  > {cmd}')

    # Guardar
    enviar(shell, 'end', 1)
    enviar(shell, 'write memory', 3)
    ssh.close()
    print(f'[OK] {nombre} configurado y guardado.')


def configurar_linux(ruta, comandos, nombre):
    """Configura un host Linux (Alpine)."""
    print(f'\n{"="*50}')
    print(f'  Configurando: {nombre}')
    print(f'{"="*50}')
    ssh, shell = abrir_consola(ruta)

    # Elevar a root
    enviar(shell, 'sudo su', 2)

    for cmd in comandos:
        enviar(shell, cmd, 2)
        print(f'  > {cmd}')

    ssh.close()
    print(f'[OK] {nombre} configurado.')


# -- Menu --
print()
print('=' * 50)
print('  Router-on-a-Stick — CML Console Server')
print('  VPN al sandbox DevNet requerida')
print('=' * 50)
print()
print('1. Solo Router (R1)')
print('2. Solo Switch (SW1)')
print('3. Solo Hosts (PC1 + PC2)')
print('4. TODO (R1 + SW1 + Hosts)')

opcion = input('\nOpcion (1-4): ').strip()

if opcion in ('1', '4'):
    configurar_cisco(CONSOLA_R1, COMANDOS_R1, 'Router R1')

if opcion in ('2', '4'):
    configurar_cisco(CONSOLA_SW1, COMANDOS_SW1, 'Switch SW1')

if opcion in ('3', '4'):
    configurar_linux(CONSOLA_PC1, COMANDOS_PC1, 'PC1 desktop-1 (VLAN 10)')
    configurar_linux(CONSOLA_PC2, COMANDOS_PC2, 'PC2 desktop-0 (VLAN 20)')

print('\n--- Listo! ---')

config.py

Datos de conexion CML, comandos del router, switch y hosts. Solo este archivo cambia si cambian IPs, VLANs o interfaces.

config.py
# Router-on-a-Stick — config.py
#
# Conexion via CML Console Server (SSH a 10.10.20.161)
# VPN activa antes de ejecutar.
#
# VLSM base: 192.168.50.0/24
#   Compras    192.168.50.0/26   GW .1    (62 hosts)
#   Ventas     192.168.50.64/27  GW .65   (30 hosts)
#   Inventario 192.168.50.96/28  GW .97   (14 hosts)
#   Gestion    192.168.50.112/30 GW .113  (2 hosts)

# -- CML Console Server --
CML_HOST = '10.10.20.161'
CML_USER = 'developer'
CML_PASS = 'C1sco12345'
ENABLE_SECRET = 'cisco'

# Rutas de consola: /NombreLab/NombreNodo/Linea
CONSOLA_R1  = '/Examen/cat8000v-0/0'
CONSOLA_SW1 = '/Examen/iosvl2-0/0'
CONSOLA_PC1 = '/Examen/desktop-1/0'
CONSOLA_PC2 = '/Examen/desktop-0/0'

# -- ROUTER R1 --
# Topologia: R1 GigabitEthernet1 <--> SW1 Gi0/0 (trunk)
COMANDOS_R1 = [
    'hostname R1',
    'no ip domain-lookup',
    'banner motd #Acceso solo para personal de TI#',

    # Interfaz fisica (trunk)
    'interface GigabitEthernet1',
    'no ip address',
    'no shutdown',
    'exit',

    # VLAN 10 - Compras
    'interface GigabitEthernet1.10',
    'encapsulation dot1Q 10',
    'ip address 192.168.50.1 255.255.255.192',
    'description GW_COMPRAS',
    'exit',

    # VLAN 20 - Ventas
    'interface GigabitEthernet1.20',
    'encapsulation dot1Q 20',
    'ip address 192.168.50.65 255.255.255.224',
    'description GW_VENTAS',
    'exit',

    # VLAN 30 - Inventario
    'interface GigabitEthernet1.30',
    'encapsulation dot1Q 30',
    'ip address 192.168.50.97 255.255.255.240',
    'description GW_INVENTARIO',
    'exit',

    # VLAN 99 - Gestion
    'interface GigabitEthernet1.99',
    'encapsulation dot1Q 99',
    'ip address 192.168.50.113 255.255.255.252',
    'description GW_GESTION',
    'exit',
]

# -- SWITCH SW1 --
# Topologia:
#   SW1 Gi0/0 <--> R1 G1       (trunk)
#   SW1 Gi0/1 <--> desktop-1   (VLAN 10)
#   SW1 Gi0/2 <--> desktop-0   (VLAN 20)
COMANDOS_SW1 = [
    'hostname SW1',
    'banner motd #Acceso solo para personal de TI#',

    # Crear VLANs
    'vlan 10', 'name COMPRAS', 'exit',
    'vlan 20', 'name VENTAS', 'exit',
    'vlan 30', 'name INVENTARIO', 'exit',
    'vlan 99', 'name GESTION', 'exit',

    # Access: desktop-1 -> VLAN 10
    'interface GigabitEthernet0/1',
    'switchport mode access',
    'switchport access vlan 10',
    'no shutdown', 'exit',

    # Access: desktop-0 -> VLAN 20
    'interface GigabitEthernet0/2',
    'switchport mode access',
    'switchport access vlan 20',
    'no shutdown', 'exit',

    # Trunk hacia R1
    'interface GigabitEthernet0/0',
    'switchport trunk encapsulation dot1q',
    'switchport mode trunk',
    'switchport trunk allowed vlan 10,20,30,99',
    'switchport nonegotiate',
    'no shutdown', 'exit',

    # SVI gestion
    'interface vlan 99',
    'ip address 192.168.50.114 255.255.255.252',
    'no shutdown', 'exit',
    'ip default-gateway 192.168.50.113',
]

# -- HOSTS LINUX (Alpine) --
# Necesitan sudo su antes (el script lo maneja)
COMANDOS_PC1 = [
    'ip addr add 192.168.50.2/26 dev eth0',
    'ip route add default via 192.168.50.1',
]

COMANDOS_PC2 = [
    'ip addr add 192.168.50.66/27 dev eth0',
    'ip route add default via 192.168.50.65',
]

Que memorizar

  • paramiko: SSHClient() → .connect(host, username, password) → .invoke_shell() → shell.send() / shell.recv()
  • CML Console Server: SSH a 10.10.20.161 con developer/C1sco12345. Comando open /Lab/Nodo/0 abre la consola serial.
  • Router subinterfaz: interface GigabitEthernet1.10 → encapsulation dot1Q 10 → ip address
  • Switch trunk: switchport trunk encapsulation dot1q → switchport mode trunk → switchport trunk allowed vlan
  • Switch access: switchport mode access → switchport access vlan 10
  • Linux IP: sudo su → ip addr add IP/CIDR dev eth0 → ip route add default via GATEWAY
